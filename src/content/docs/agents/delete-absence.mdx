import {Steps} from "@astrojs/starlight/components";

<Steps>

1. Navigate to the absences section of the dashboard, search for the absence you want to delete and open the context menu for it. Click on _Delete_.

   ![Empty absences list](./assets/delete-absence/1-absences-list-with-context-menu.png)

1. A modal opens. Click _Delete Absence_

   ![Delete absence form](./assets/delete-absence/2-delete-absence-form.png)

1. The absence will be removed.

   ![Absence deleted](./assets/delete-absence/3-absence-deleted.png)

</Steps>
